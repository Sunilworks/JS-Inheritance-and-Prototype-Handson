Q.2) Write code to explain prototype chaining 

        const Gfather = {
            sayHello : function(){
                console.log('Hey GrandFather!!!');
            }
        }
    
        const Father = Object.create(Gfather);
            Father.Hello = function(){
              console.log('Say Hi to father');
            }
          
        const You = Object.create(Father);
            You.Hi = function(){
                console.log('Say Hi to Me!!')
            }
            
        You.sayHello();                              //accessing from the grandparent object
        You.Hello();                                 //accessing from the Parent object

        
        OUTPUT:
        Hey GrandFather!!!
        Say Hi to father

Q.3) Add a method to calculate sum of all elements in Array in array's prototype, use that method to calculate sum for multiple arrays 

        let newArr = new Array(9,37,269);
        newArr.__proto__.add = function(arr){
            let count=0;
            for(let i=0; i < arr.length; i++)
            {
                count += arr[i];
            }
            return count;
        }

        let ans = newArr.add(newArr);
        console.log(ans);
        
        
        OUTPUT:
        315
